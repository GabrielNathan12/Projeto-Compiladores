funcao inteiro Fibonacci(inteiro n){
    inteiro fibo = 0;
    inteiro antes = 0;

    para(inteiro i = 1; i <= n; i= i+1){
        se(i == 1){
            fibo = 1;
            antes = 0;
        }naose{
            fibo += antes;
            antes = fibo - antes;
        }
    }

    
    retorna fibo;
}

    ## Comenta apenas uma linha do trecho do código
    ## outputql significa saída com quebra de linha
    ## output significa saída sem quebra de linha

    inteiro n = fibo;

        soma += Fibonacci(i);
    }

}
